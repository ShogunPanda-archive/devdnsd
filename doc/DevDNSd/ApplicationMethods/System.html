def action_start
  get_logger.info(i18n.starting)

  if !Process.respond_to?(:fork) then
    logger.warn(i18n.no_fork)
    @config.foreground = true
  end

  @config.foreground ? perform_server : RExec::Daemon::Controller.start(self.class)
  true
end
